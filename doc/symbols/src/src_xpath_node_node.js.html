 15 /**
 16  * A class which implements and extends the xrx.xdm interface.
 17  * 
 18  * @constructor
 19  * @implements {xrx.xdm}
 20  */
 21 xrx.node = function(type, token, instance) {
 22   goog.base(this);
 23 
 24   /**
 25    * @type {xrx.node}
 26    * @private
 27    */
 28   this.type_ = type;
 29 
 30   /**
 31    * @type {xrx.token}
 32    * @private
 33    */
 34   this.token_ = token;
 35 
 36   /**
 37    * @type {xrx.instance}
 38    * @private
 39    */
 40   this.instance_ = instance;
 41 };
 42 goog.inherits(xrx.node, xrx.xdm);

 91 /**
 92  * Returns the string-value of the required type from a node.
 93  *
 94  * @param {!xrx.node} node The node to get value from.
 95  * @return {string} The value required.
 96  */
 97 xrx.node.prototype.getValueAsString = function() {
 98   return this.stringValue();
 99 };
100 
101 
102 
103 /**
104  * Returns the string-value of the required type from a node, casted to number.
105  *
106  * @param {!xrx.node} node The node to get value from.
107  * @return {number} The value required.
108  */
109 xrx.node.prototype.getValueAsNumber = function() {
110   return +this.getValueAsString();
111 };
112 
113 
114 
115 /**
116  * Returns the string-value of the required type from a node, casted to boolean.
117  *
118  * @param {!xrx.node} node The node to get value from.
119  * @return {boolean} The value required.
120  */
121 xrx.node.prototype.getValueAsBool = function() {
122   return !!this.getValueAsString();
123 };
