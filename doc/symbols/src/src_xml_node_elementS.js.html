162 xrx.node.ElementS.prototype.getName = function() {
163   var xml = this.getInstance().xml();
164   var loc = this.getInstance().getStream().tagName(
165       this.getToken().xml(xml));
166   loc.offset += this.getToken().offset();
167 
168   return loc.xml(xml);
169 };

185 xrx.node.ElementS.prototype.getStringValue = function() {
186   var xml = this.instance_.xml();
187   var traverse = new xrx.traverse(xml);
188   var string = '';
189   var self = this;
190 
191   traverse.rowStartTag = function(label, offset, length1, length2) {
192     string += xml.substr(offset + length1, length2 - length1);
193   };
194 
195   traverse.rowEmptyTag = function(label, offset, length1, length2) {
196     string += xml.substr(offset + length1, length2 - length1);
197   };
198 
199   traverse.rowEndTag = function(label, offset, length1, length2) {
200     var stop = self.getLabel().sameAs(label);
201     if (!stop) string += xml.substr(offset + length1, length2 - length1);
202     if (stop) traverse.stop();
203   };
204 
205   traverse.forward(this.getToken());
206 
207   return string;
208 };

212 xrx.node.ElementS.prototype.getXml = function() {
213   if (this.getToken().type() === xrx.token.EMPTY_TAG) {
214     return this.getToken().xml(this.instance_.xml());
215   }
216 
217   var pilot = new xrx.pilot(this.instance_.xml());
218   var target = new xrx.token.EndTag(this.getLabel().clone());
219   var endTag = pilot.location(this.getToken(), target);
220 
221   return this.instance_.xml().substring(this.getToken().offset(),
222       endTag.offset() + endTag.length());
223 };

303 /**
304  * @param {!xrx.label}
305  */
306 xrx.node.ElementS.prototype.forward = function(stop) {
307   var self = this;
308   var traverse = new xrx.traverse(this.instance_.xml());
309   var token;
310 
311   traverse.rowStartTag = function(label, offset, length1, length2) {
312     var tag = new xrx.token.StartTag(label.clone(), offset, length1);
313     self.eventNode(new xrx.node.ElementS(self.instance_, tag));
314     if (length1 !== length2) {
315       self.eventNode(new xrx.node.TextS(self.instance_, tag, length2 - length1));
316     }
317   };
318 
319   traverse.rowEmptyTag = function(label, offset, length1, length2) {
320     var tag = new xrx.token.EmptyTag(label.clone(), offset, length1);
321 
322     self.eventNode(new xrx.node.ElementS(self.instance_, tag));
323     if (length1 !== length2) {
324       self.eventNode(new xrx.node.TextS(self.instance_, tag, length2 - length1));
325     }
326   };
327 
328   traverse.rowEndTag = function(label, offset, length1, length2) {
329     var tag = new xrx.token.EndTag(label.clone(), offset, length1);
330 
331     if (length1 !== length2) {
332       self.eventNode(new xrx.node.TextS(self.instance_, tag, length2 - length1));
333     }
334     if (label.sameAs(stop)) traverse.stop();
335   };
336 
337   traverse.forward(self.getToken());
338 };

342 /**
343  * @param {!xrx.label}
344  */
345 xrx.node.ElementS.prototype.backward = function(stop) {
346   var self = this;
347   var traverse = new xrx.traverse(this.instance_.xml());
348   var token;
349 
350   traverse.rowStartTag = function(label, offset, length1, length2) {
351     var tag = new xrx.token.StartTag(label.clone(), offset, length1);
352 
353     if (length1 !== length2) {
354       //self.eventNode(new xrx.node.TextS(self.instance_, tag, length2 - length1));
355     }
356     self.eventNode(new xrx.node.ElementS(self.instance_, tag));
357     if (label.sameAs(stop)) traverse.stop();
358   };
359 
360   traverse.rowEmptyTag = function(label, offset, length1, length2) {
361     var tag = new xrx.token.EmptyTag(label.clone(), offset, length1);
362 
363     if (length1 !== length2) {
364       //self.eventNode(new xrx.node.TextS(self.instance_, tag, length2 - length1));
365     }
366     self.eventNode(new xrx.node.ElementS(self.instance_, tag));
367     if (label.sameAs(stop)) traverse.stop();
368   };
369 
370   traverse.rowEndTag = function(label, offset, length1, length2) {
371     var tag = new xrx.token.EndTag(label.clone(), offset, length1);
372 
373     if (length1 !== length2) {
374       //self.eventNode(new xrx.node.TextS(self.instance_, tag, length2 - length1));
375     }
376   };
377   traverse.backward(self.getToken());
378 };
