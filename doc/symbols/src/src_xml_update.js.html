 24 /**
 25  * Shared function for all replace operations.
 26  * @private
 27  */
 28 xrx.update.replace_ = function(instance, token, xml) {
 29   var diff = xml.length - token.length();
 30 
 31   instance.update(token.offset(), token.length(), xml);
 32 
 33   token.length(xml.length);
 34 
 35   return diff;
 36 };

116  * @param {!xrx.token.NotTag} target The not-tag token where the empty tag is inserted.
117  * @param {!integer} offset The offset relative to the not-tag token.
118  * @param {!string} localName The local name of the new token.
119  * @param {!string} opt_namespaceUri The namespace URI of the new token.

121 xrx.update.insertEmptyTag = function(instance, target, offset, localName,
122     opt_namespaceUri) {
123   var diff;
124

126     diff = xrx.update.insert_(instance, target.offset() + offset,

128   } else {
129     var nsPrefix = instance.getIndex().getNamespacePrefix(target, opt_namespaceUri);
130 
131     diff = xrx.update.insert_(instance, target.offset() + offset,
132         xrx.serialize.emptyTagNs(nsPrefix, localName, opt_namespaceUri));
133 
134     //TODO: add namespace declaration to index
135   }
136 
137   return diff;

142 /**
143  * Wrap a piece of XML with a start-tag and a end-tag.
144  * 
145  * @param {!xrx.instance} instance The instance to be updated.
146  * @param {!xrx.token.NotTag} target1 The left not-tag token where the new
147  * start-tag shall be inserted.
148  * @param {!xrx.token.NotTag} target2 The right not-tag token where the
149  * new end-tag shall be inserted.
150  * @param {!integer} offset1 The offset relative to the left not-tag token.
151  * @param {!integer} offset2 The offset relative to the right not-tag token.
152  * @param {!string} localName The local name of the new token.
153  * @param {!string} opt_namespaceUri The namespace URI of the new token.
154  */
155 xrx.update.insertStartEndTag = function(instance, target1, target2, offset1, offset2,
156     localName, opt_namespaceUri) {
157   var diffs;
158   var diff2;
159 
160   if (!opt_namespaceUri) {
161 
162     diff1 = xrx.update.insert_(instance, target2.offset() + offset2,
163         xrx.serialize.endTag(localName));
164     diff2 = xrx.update.insert_(instance, target1.offset() + offset1,
165         xrx.serialize.startTag(localName));
166   } else {
167     var nsPrefix = instance.getIndex().getNamespacePrefix(target1, opt_namespaceUri);
168 
169     diff1 = xrx.update.insert_(instance, target2.offset() + offset2,
170         xrx.serialize.endTagNs(nsPrefix, localName, opt_namespaceUri));
171 
172     diff2 = xrx.update.insert_(instance, target1.offset() + offset1,
173         xrx.serialize.startTagNs(nsPrefix, localName, opt_namespaceUri));
174 
175     //TODO: add namespace declaration to index
176   }
177 
178   return [diff1, diff2];
179 };

197 /**
198  * Inserts a new attribute into a start-tag or a empty tag.
199  * 
200  * @param {!xrx.instance} instance The instance to be updated.
201  * @param {!(xrx.token.StartTag|xrx.token.EmptyTag)} target The tag where the attribute 
202  * shall be inserted.
203  * @param {!string} qName The qualified name of the new attribute.
204  * @param {!string} opt_namespaceUri The namespace URI of the new attribute.
205  */
206 xrx.update.insertAttribute = function(instance, parent, qName,
207     opt_namespaceUri) {
208   var diff;
209   var loc = instance.getStream().tagName(parent.xml(instance.xml()));
210 
211   if (!opt_namespaceUri) {
212     diff = xrx.update.insert_(instance, parent.offset() + loc.offset +
213         loc.length, xrx.serialize.attribute(qName, ''));
214   } else {
215     var nsPrefix1 = instance.getIndex().getNamespacePrefix(parent, opt_namespaceUri);
216     var nsPrefix2 = qName.split(':')[0];
217     if (nsPrefix1 !== 'xmlns:' + nsPrefix2 && nsPrefix1 !== undefined &&
218         nsPrefix1 !== 'xmlns') throw Error('Prefix ' + nsPrefix2 +
219         ' is not bound to namespace ' + opt_namespaceUri + '.');
220 
221     diff = xrx.update.insert_(instance, parent.offset() + loc.offset +
222         loc.length, xrx.serialize.attributeNs(nsPrefix1, qName, opt_namespaceUri));
223 
224     //TODO: add namespace declaration to index
225   }
226 
227   return diff;
228 };
