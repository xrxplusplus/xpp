 97 xrx.node.DocumentS.prototype.getStringValue = function() {
 98   var xml = this.instance_.xml();
 99   var traverse = new xrx.traverse(xml);
100   var string = '';
101   var self = this;
102 
103   traverse.rowStartTag = function(label, offset, length1, length2) {
104     string += xml.substr(offset + length1, length2 - length1);
105   };
106 
107   traverse.rowEmptyTag = function(label, offset, length1, length2) {
108     string += xml.substr(offset + length1, length2 - length1);
109   };
110 
111   traverse.rowEndTag = function(label, offset, length1, length2) {
112     string += xml.substr(offset + length1, length2 - length1);
113   };
114 
115   traverse.forward();
116 
117   return string;
118 };

170 /**
171  * @param {!xrx.label}
172  */
173 xrx.node.DocumentS.prototype.forward = function(stop) {
174   var self = this;
175   var traverse = new xrx.traverse(this.instance_.xml());
176   var token;
177 
178   traverse.rowStartTag = function(label, offset, length1, length2) {
179     var tag = new xrx.token.StartTag(label.clone(), offset, length1);
180     self.eventNode(new xrx.node.ElementS(self.instance_, tag));
181     if (length1 !== length2) {
182       self.eventNode(new xrx.node.TextS(self.instance_, tag, length2 - length1));
183     }
184     if (label.sameAs(stop)) traverse.stop();
185   };
186 
187   traverse.rowEmptyTag = function(label, offset, length1, length2) {
188     var tag = new xrx.token.EmptyTag(label.clone(), offset, length1);
189 
190     self.eventNode(new xrx.node.ElementS(self.instance_, tag));
191     if (length1 !== length2) {
192       self.eventNode(new xrx.node.TextS(self.instance_, tag, length2 - length1));
193     }
194     if (label.sameAs(stop)) traverse.stop();
195   };
196 
197   traverse.rowEndTag = function(label, offset, length1, length2) {
198     var tag = new xrx.token.EndTag(label.clone(), offset, length1);
199 
200     if (length1 !== length2) {
201       self.eventNode(new xrx.node.TextS(self.instance_, tag, length2 - length1));
202     }
203   };
204 
205   traverse.forward();
206 };
