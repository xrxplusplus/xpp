 14 /**
 15  * Constructs a new not-tag token.
 16  * xrx.token.NotTag is a container token for all tokens which
 17  * are no tags and not part of a tag (attribute, namespace), 
 18  * i.e. text, comment, processing instruction.
 19  *  
 20  * @constructor
 21  * @extends xrx.token
 22  */
 23 xrx.token.NotTag = function(label, opt_offset, opt_length) {
 24   goog.base(this, xrx.token.NOT_TAG, label, opt_offset, opt_length);
 25 };
