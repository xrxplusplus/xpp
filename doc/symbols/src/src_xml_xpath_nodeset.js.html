 12 /**
 13  * A set of nodes sorted by their prefix order in the document.
 14  *
 15  * @constructor
 16  */
 17 xrx.xpath.NodeSet = function() {
 18   // In violation of standard Closure practice, we initialize properties to
 19   // immutable constants in the constructor instead of on the prototype,
 20   // because we have empirically measured better performance by doing so.
 21 
 22   /**
 23    * A pointer to the first node in the linked list.
 24    *
 25    * @private
 26    * @type {xrx.xpath.NodeSet.Entry_}
 27    */
 28   this.first_ = null;
 29 
 30   /**
 31    * A pointer to the last node in the linked list.
 32    *
 33    * @private
 34    * @type {xrx.xpath.NodeSet.Entry_}
 35    */
 36   this.last_ = null;
 37 
 38   /**
 39    * Length of the linked list.
 40    *
 41    * @private
 42    * @type {number}
 43    */
 44   this.length_ = 0;
 45 };

 78 /**
 79  * Merges two node-sets, removing duplicates. This function may modify both
 80  * node-sets, and will return a reference to one of the two.
 81  *
 82  * <p> Note: We assume that the two node-sets are already sorted in DOM order.
 83  *
 84  * @param {!xrx.xpath.NodeSet} a The first node-set.
 85  * @param {!xrx.xpath.NodeSet} b The second node-set.
 86  * @return {!xrx.xpath.NodeSet} The merged node-set.
 87  */
 88 xrx.xpath.NodeSet.merge = function(a, b) {
 89   if (!a.first_) {
 90     return b;
 91   } else if (!b.first_) {
 92     return a;
 93   }
 94   var aCurr = a.first_;
 95   var bCurr = b.first_;
 96   var merged = a, tail = null, next = null, length = 0;
 97   while (aCurr && bCurr) {
 98     if (aCurr.node.isSameAs(bCurr.node)) {
 99       next = aCurr;
100       aCurr = aCurr.next;
101       bCurr = bCurr.next;
102     } else {
103       if (aCurr.node.isAfter(bCurr.node)) {
104         next = bCurr;
105         bCurr = bCurr.next;
106       } else {
107         next = aCurr;
108         aCurr = aCurr.next;
109       }
110     }
111     next.prev = tail;
112     if (tail) {
113       tail.next = next;
114     } else {
115       merged.first_ = next;
116     }
117     tail = next;
118     length++;
119   }
120   next = aCurr || bCurr;
121   while (next) {
122     next.prev = tail;
123     tail.next = next;
124     tail = next;
125     length++;
126     next = next.next;
127   }
128   merged.last_ = tail;
129   merged.length_ = length;
130   return merged;
131 };

134 /**
135  * Prepends a node to this node-set.
136  *
137  * @param {!xrx.node} node The node to be added.
138  */
139 xrx.xpath.NodeSet.prototype.unshift = function(node) {
140   var entry = new xrx.xpath.NodeSet.Entry_(node);
141   entry.next = this.first_;
142   if (!this.last_) {
143     this.first_ = this.last_ = entry;
144   } else {
145     this.first_.prev = entry;
146   }
147   this.first_ = entry;
148   this.length_++;
149 };

152 /**
153  * Adds a node to this node-set.
154  *
155  * @param {!xrx.node} node The node to be added.
156  */
157 xrx.xpath.NodeSet.prototype.add = function(node) {
158   var entry = new xrx.xpath.NodeSet.Entry_(node);
159   entry.prev = this.last_;
160   if (!this.first_) {
161     this.first_ = this.last_ = entry;
162   } else {
163     this.last_.next = entry;
164   }
165   this.last_ = entry;
166   this.length_++;
167 };

172  *
173  * @return {?xrx.node} The first node of the nodeset
174                                      if the nodeset is non-empty;
175  *     otherwise null.
176  */
177 xrx.xpath.NodeSet.prototype.getFirst = function() {
178   var first = this.first_;
179   if (first) {
180     return first.node;
181   } else {
182     return null;
183   }

220  *     add to this nodeset until the iterator is done.

231 /**
232  * An iterator over the nodes of this nodeset.
233  *
234  * @param {!xrx.xpath.NodeSet} nodeset The nodeset to be iterated over.
235  * @param {boolean} reverse Whether to iterate in ascending or descending
236  *     order.
237  * @constructor
238  */
239 xrx.xpath.NodeSet.Iterator = function(nodeset, reverse) {
240   // In violation of standard Closure practice, we initialize properties to
241   // immutable constants in the constructor instead of on the prototype,
242   // because we have empirically measured better performance by doing so.
243 
244   /**
245    * @type {!xrx.xpath.NodeSet}
246    * @private
247    */
248   this.nodeset_ = nodeset;
249 
250   /**
251    * @type {boolean}
252    * @private
253    */
254   this.reverse_ = reverse;
255 
256   /**
257    * @type {xrx.xpath.NodeSet.Entry_}
258    * @private
259    */
260   this.current_ = reverse ? nodeset.last_ : nodeset.first_;
261 
262   /**
263    * @type {xrx.xpath.NodeSet.Entry_}
264    * @private
265    */
266   this.lastReturned_ = null;
267 };
268 
269 
270 /**
271  * Returns the next value of the iteration or null if passes the end.
272  *
273  * @return {?xrx.node} The next node from this iterator.
274  */
275 xrx.xpath.NodeSet.Iterator.prototype.next = function() {
276   var current = this.current_;
277   if (current == null) {
278     return null;
279   } else {
280     var lastReturned = this.lastReturned_ = current;
281     if (this.reverse_) {
282       this.current_ = current.prev;
283     } else {
284       this.current_ = current.next;
285     }
286     return lastReturned.node;
287   }
288 };

291 /**
292  * Deletes the last node that was returned from this iterator.
293  */
294 xrx.xpath.NodeSet.Iterator.prototype.remove = function() {
295   var nodeset = this.nodeset_;
296   var entry = this.lastReturned_;
297   if (!entry) {
298     throw Error('Next must be called at least once before remove.');
299   }
300   var prev = entry.prev;
301   var next = entry.next;
302 
303   // Modify the pointers of prev and next
304   if (prev) {
305     prev.next = next;
306   } else {
307     // If there was no prev node entry must've been first_, so update first_.
308     nodeset.first_ = next;
309   }
310   if (next) {
311     next.prev = prev;
312   } else {
313     // If there was no prev node entry must've been last_, so update last_.
314     nodeset.last_ = prev;
315   }
316   nodeset.length_--;
317   this.lastReturned_ = null;
318 };
