 47 xrx.richxml.placeholder = {
 48   startTag: unescape('%BB'),
 49   endTag: unescape('%AB'),
 50   emptyTag: unescape('%D7')
 51 };

 55 /**
 56  *
 57  */
 58 xrx.richxml.placeholder.matches = function(ch) {
 59 
 60   for(var p in xrx.richxml.placeholder) {
 61     if (xrx.richxml.placeholder[p] === ch) {
 62       return true;
 63     }
 64   }
 65   return false;
 66 };

 96 /**
 97  * @private
 98  */
 99 xrx.richxml.prototype.transformXml_ = function(xml) {
100   var visualXml = '';
101   var self = this;
102   var stream = new xrx.stream(xml);
103 
104   stream.rowStartTag = function(offset, length1, length2) {
105     visualXml += xrx.richxml.placeholder.startTag;
106     if(length1 !== length2) visualXml += xml.substr(offset + length1, length2 - length1);
107   };
108 
109   stream.rowEndTag = function(offset, length1, length2) {
110     visualXml += xrx.richxml.placeholder.endTag;
111     if(length1 !== length2) visualXml += xml.substr(offset + length1, length2 - length1);
112   };
113 
114   stream.rowEmptyTag = function(offset, length1, length2) {
115     visualXml += xrx.richxml.placeholder.emptyTag;
116     if(length1 !== length2) visualXml += xml.substr(offset + length1, length2 - length1);
117   };
118 
119   stream.forward();
120 
121   return visualXml;
122 };

166 /**
167  *
168  */
169 xrx.richxml.prototype.markActiveTags = function(unmarkflag) {
170 
171   var cm = this.codemirror_;
172   var cursor = xrx.ui.wysiwym.cursor;
173   var self = this;
174   
175   var markedElements = function() { 
176     return self.markedElements_ || (self.markedElements_ = []); 
177   };
178 
179   var correspondingTag = function() {
180     var text = cm.getValue();
181     var index = cm.indexFromPos(cm.getCursor()) - 1;
182     var ch;
183     var stack = 0;
184 
185     if(cursor.leftTokenOutside(self).string === xrx.richxml.placeholder.startTag) {
186 
187       for(var i = index; i < text.length; i++) {
188         ch = text[i];
189         if(ch === xrx.richxml.placeholder.endTag) stack -= 1;
190         if(ch === xrx.richxml.placeholder.startTag) stack += 1;
191         if(stack === 0) {
192           index = i + 1;
193           break;
194         } 
195       }
196     } else if(cursor.leftTokenOutside(self).string === xrx.richxml.placeholder.endTag) {
197 
198       for(var i = index; i >= 0; i--) {
199         ch = text[i];
200         if(ch == xrx.richxml.placeholder.endTag) stack += 1;
201         if(ch == xrx.richxml.placeholder.startTag) stack -= 1;
202         if(stack == 0) {
203           index = i + 1;
204           break;
205         } 
206       }
207     } else {};
208 
209     return cm.posFromIndex(index);
210   };
211   
212   var mark = function(from, to, style) {
213     var mark = cm.markText(from, to, style);
214 
215     markedElements().push(mark);
216   };
217   
218   var unmark = function() {
219 
220     for(var i = 0; i < markedElements().length; ++i) {
221       markedElements()[i].clear();
222     }
223     self.markedElements_ = [];
224   };
225 
226   unmark();
227   if (!this.cursorIsNearTag()) return;
228   if (unmarkflag) return;
229 
230   var firstMark = cursor.leftTokenOutside(self).string === xrx.richxml.placeholder.startTag ? 
231       xrx.richxml.className.startTagActive : xrx.richxml.className.endTagActive;
232   var secondMark = cursor.leftTokenOutside(self).string === xrx.richxml.placeholder.startTag ? 
233       xrx.richxml.className.endTagActive : xrx.richxml.className.startTagActive;
234 
235   // mark matched element
236   mark(
237       { line: cursor.leftPosition(self).line, ch: cursor.leftPosition(self).ch - 1 }, 
238       { line: cursor.leftPosition(self).line, ch: cursor.leftPosition(self).ch }, 
239       { className: firstMark }
240   );
241   
242   if (cursor.leftTokenOutside(self).string === xrx.richxml.placeholder.emptyTag) return;
243   
244   // mark corresponding element
245   var corresponding = correspondingTag();
246   mark(
247       { line: corresponding.line, ch: corresponding.ch - 1 },
248       { line: corresponding.line, ch: corresponding.ch },
249       { className: secondMark }
250   );
251   
252   // mark text
253   var from = { line: cursor.leftPosition(self).line, ch: cursor.leftPosition(self).ch };
254   var to = { line: corresponding.line, ch: corresponding.ch - 1 };
255   cursor.leftTokenOutside(self).string === xrx.richxml.placeholder.startTag ? 
256       mark(from, to, { className: xrx.richxml.className.notTagActive }) : 
257       mark(to, from, { className: xrx.richxml.className.notTagActive });
258 };

354 /**
355  * 
356  */
357 xrx.richxml.prototype.eventCursorActivity = function() {
358   var cm = this.codemirror_;
359 
360   if (cm.somethingSelected()) {
361     this.setReadonly(true);
362     this.cursorActivitySelection();
363   } else {
364     this.setReadonly(false);
365     this.markActiveTags();
366     this.showElements();
367     this.showAttributes();
368     this.showTagName();
369   }
370 };
