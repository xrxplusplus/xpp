  6 goog.provide('xrx.token.Tag');

 10 goog.require('xrx.token');

 15 /**
 16  * Constructs a new tag token. The tag token is a generic 
 17  * container token for all kinds of native tag tokens as 
 18  * well as all generic tag tokens.
 19  * 
 20  * @constructor
 21  * @extends xrx.token
 22  */
 23 xrx.token.Tag = function(label) {
 24   goog.base(this, xrx.token.TAG, label);
 25 };
 26 goog.inherits(xrx.token.Tag, xrx.token.Abstract);
 27 
 28 
 29 
 30 /**
 31  * Compares the generic type of two tokens.
 32  *
 33  * @param {!number} type The type to check against.
 34  * @return {!boolean}
 35  */
 36 xrx.token.Tag.prototype.typeOf = function(type) {
 37   return this.type_ === type || xrx.token.START_TAG === type
 38       || xrx.token.END_TAG === type || xrx.token.EMPTY_TAG === type || 
 39       xrx.token.START_EMPTY_TAG === type;
 40 };
