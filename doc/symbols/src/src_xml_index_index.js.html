 21 /**
 22  * A class to create long-lived binary encodings for XML instances.
 23  * 
 24  * @param {string} The XML string to be indexed.
 25  * @constructor
 26  */
 27 xrx.index = function(xml) {
 28 
 29   /**
 30    * The indexed XML rows.
 31    * @type {Array.<xrx.index.row>}
 32    */
 33   this.rows_ = [];
 34 
 35   /**
 36    * Current position of row iteration.
 37    * @type {integer}
 38    */
 39   this.iterKey_ = 0;
 40 
 41   /**
 42    * Indexed namespace table.
 43    * @type {Array.<xrx.index.Namespace>}
 44    */
 45   this.tNamespace_ = [];
 46 
 47 
 48 
 49   this.build(xml);
 50 };

 74 /**
 75  * Searches a index key by overloading a tag token.
 76  * 
 77  * @param {xrx.token} token The tag token.
 78  * @param {integer} opt_start The index key where to start.
 79  * @param {boolean} opt_reverse False to search in forward or true to search in 
 80  * backward direction.
 81  * @return {integer} The key.
 82  */
 83 xrx.index.prototype.getKeyByTag = function(token, opt_start) {
 84   var row;
 85   var key;
 86 
 87   opt_start ? this.iterSetKey(opt_start) : this.iterAtFirst();
 88 
 89   do {
 90     row = this.iterGetRow();
 91     key = this.iterGetKey();
 92 
 93     if (token.typeOf(row.getType()) && 
 94         this.getLabel(key).sameAs(token.label())) break;
 95 
 96   } while (row = this.iterNext());
 97 
 98   return key;
 99 };

103 /**
104  * Searches a index key by overloading a not-tag token.
105  * @param {xrx.token} token The not-tag token.
106  * @return {integer} The key.
107  */
108 xrx.index.prototype.getKeyByNotTag = function(token, opt_start) {
109   var row;
110   var key;
111   var isStartTagRow = token.label().last() === 0 ? true : false;
112 
113   opt_start ? this.iterSetKey(opt_start) : this.iterAtFirst();
114 
115   if (isStartTagRow) {
116     var searchLabel = token.label().clone();
117     searchLabel.parent();
118 
119     do {
120       row = this.iterGetRow();
121       key = this.iterGetKey();
122   
123       if (row.getType() === xrx.token.START_TAG && 
124           this.getLabel(key).sameAs(searchLabel)) break;
125   
126     } while (row = this.iterNext());
127     
128   } else {
129     var searchLabel = token.label().clone();
130 
131     do {
132       row = this.iterGetRow();
133       key = this.iterGetKey();
134   
135       if ((row.getType() === xrx.token.EMPTY_TAG || row.getType() === xrx.token.END_TAG) && 
136           this.getLabel(key).sameAs(searchLabel)) break;
137   
138     } while (row = this.iterNext());
139   }
140 
141   return key;
142 };

181 /**
182  * Returns a row label by overloading a index key.
183  * @return {xrx.label} The label.
184  */
185 xrx.index.prototype.getLabel = function(key) {
186   var row = this.rows_[key];
187   var next;
188   var last;
189   var label = [];
190 
191   label.unshift(row.getPosition());
192   if (key === 0 || key === this.getLastKey()) return new xrx.label(label);
193 
194   for(;;) {
195     next = row.getParent();
196     row = this.rows_[next];
197     label.unshift(row.getPosition());
198     if (next === 0 || next === last) break;
199     last = next; // to avoid endless loop
200   }
201 
202   return new xrx.label(label);
203 };

207 /**
208  * Returns a tag token by overloading a row.
209  * @return {xrx.token.StartTag|xrx.token.EmptyTag|xrx.token.EndTag} The tag. 
210  */
211 xrx.index.prototype.getTag = function(row) {
212   var r = this.rows_[row];
213   var tag = new xrx.token.Abstract(r.getType(), this.getLabel(row),
214       r.getOffset(), r.getLength1());
215   
216   return xrx.token.native(tag);
217 };

295 /**
296  * A namespace object. Namespace tokens as the only XML tokens
297  * are statically extracted during index building.
298  *
299  * @param {xrx.label} opt_label The label of the namespace token.
300  * @param {string} opt_prefix The namespace prefix.
301  * @param {string} opt_uri The namespace URI.
302  */
303 xrx.index.Namespace = function(opt_label, opt_prefix, opt_uri) {
304 
305 
306 
307   this.label = opt_label;
308 
309 
310 
311   this.parentLabel;
312 
313 
314 
315   this.prefix = opt_prefix;
316 
317 
318 
319   this.uri = opt_uri;
320 };
321

324 /**
325  * @return {xrx.index.Namespace|undefined}
326  */
327 xrx.index.prototype.getNamespace = function(token, prefix) {
328 
329   return goog.array.findRight(this.tNamespace_, function(ns, ind, arr) {
330     return ns.prefix === prefix && (token.label().sameAs(ns.parentLabel) ||
331         token.label().isDescendantOf(ns.parentLabel));
332   });
333 };
334 
335 
336 
337 /**
338  * @return {string|undefined}
339  */
340 xrx.index.prototype.getNamespaceUri = function(token, prefix) {
341 
342   var namespace = goog.array.findRight(this.tNamespace_, function(ns, ind, arr) {
343     return ns.prefix === prefix && (token.label().sameAs(ns.parentLabel) ||
344         token.label().isDescendantOf(ns.parentLabel));
345   });
346   
347   return namespace ? namespace.uri : undefined;
348 };

355 xrx.index.prototype.getNamespacePrefix = function(token, uri) {

379 /**
380  * Builds an index from scratch.
381  * @param {!string} The XML string.
382  */
383 xrx.index.prototype.build = function(xml) {
384   var traverse = new xrx.traverse(xml);
385   var row;
386   var index = this;
387   var parent;
388   var labelBuffer = {};
389 
390   traverse.setFeature('NS_PREFIX', true);
391   traverse.setFeature('NS_URI', true);
392   
393   var update = function(row, type, label, offset, length1, length2) {
394     parent = label.clone();
395     parent.parent();
396 
397     row.setType(type);
398     row.setPosition(label.last());
399     row.setParent(labelBuffer[parent.toString()]);
400     row.setOffset(offset);
401     row.setLength1(length1);
402     row.setLength2(length2);
403   };
404 
405   traverse.rowStartTag = function(label, offset, length1, length2) {
406 
407     update(index.head(), xrx.token.START_TAG, label, offset, length1, length2);
408     labelBuffer[label.toString()] = index.getLastKey();
409   };
410 
411   traverse.rowEmptyTag = function(label, offset, length1, length2) {
412 
413     update(index.head(), xrx.token.EMPTY_TAG, label, offset, length1, length2);
414   };
415 
416   traverse.rowEndTag = function(label, offset, length1, length2) {
417 
418     update(index.head(), xrx.token.END_TAG, label, offset, length1, length2);
419     delete labelBuffer[label.toString()];
420   };
421 
422   traverse.eventNsPrefix = function(label, offset, length) {
423     var ns = new xrx.index.Namespace(label.clone(), xml.substr(offset, length));
424     var parent = label.clone();
425     parent.parent();
426     ns.parentLabel = parent;
427     index.tNamespace_.push(ns);
428   };
429 
430   traverse.eventNsUri = function(label, offset, length) {
431     index.tNamespace_[index.tNamespace_.length - 1].uri = xml.substr(offset, length);
432   };
433 
434   traverse.forward();
435 };
