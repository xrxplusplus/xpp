 95 /**
 96  * Converts a generic token into its native form.
 97  *
 98  * @param {!xrx.token} token The token to convert.
 99  * @return {?}
100  */
101 xrx.token.native = function(token) {
102   var newToken;
103   var label = token.label().clone();
104   var offset = token.offset();
105   var length = token.length();
106   
107   var convert = function(constr) {
108     var tmp = new constr();
109     tmp.label(label);
110     tmp.offset(offset);
111     tmp.length(length);
112     return tmp;
113   };
114   
115   switch(token.type()) {
116   case xrx.token.START_TAG:
117     newToken = convert(xrx.token.StartTag);
118     break;
119   case xrx.token.END_TAG:
120     newToken = convert(xrx.token.EndTag);
121     break;
122   case xrx.token.EMPTY_TAG:
123     newToken = convert(xrx.token.EmptyTag);
124     break;
125   case xrx.token.NOT_TAG:
126     newToken = convert(xrx.token.NotTag);
127     break;
128   case xrx.token.TAG_NAME:
129     newToken = convert(xrx.token.TagName);
130     break;
131   case xrx.token.ATTRIBUE:
132     newToken = convert(xrx.token.Attribute);
133     break;
134   case xrx.token.ATTR_NAME:
135     newToken = convert(xrx.token.AttrName);
136     break;
137   case xrx.token.ATTR_VALUE:
138     newToken = convert(xrx.token.AttrValue);
139     break;
140   default:
141     throw Error('Token is generic or unknown.');
142     break;  
143   };
144   
145   return newToken;
146 };
