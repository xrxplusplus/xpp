 51 /**
 52  * 
 53  */
 54 xrx.node.TextS.prototype.getLabel = function() {
 55   var label = this.tag_.label().clone();
 56   if (this.tag_.type() === xrx.token.START_TAG) {
 57     label.push0();
 58   } else if (this.tag_.type() === xrx.token.END_TAG) {
 59     var tmp = label.pop();
 60     label.push(tmp + .5);
 61   } else {};
 62 
 63   return label;
 64 };

275 /**
276  * @param {!xrx.label}
277  */
278 xrx.node.TextS.prototype.forward = function(stop) {
279   var self = this;
280   var traverse = new xrx.traverse(this.instance_.xml());
281   var token;
282 
283   traverse.rowStartTag = function(label, offset, length1, length2) {
284     var tag = new xrx.token.StartTag(label.clone(), offset, length1);
285     self.eventNode(new xrx.node.ElementS(self.instance_, tag));
286     if (length1 !== length2) {
287       self.eventNode(new xrx.node.TextS(self.instance_, tag, length2 - length1));
288     }
289   };
290 
291   traverse.rowEmptyTag = function(label, offset, length1, length2) {
292     var tag = new xrx.token.EmptyTag(label.clone(), offset, length1);
293 
294     self.eventNode(new xrx.node.ElementS(self.instance_, tag));
295     if (length1 !== length2) {
296       self.eventNode(new xrx.node.TextS(self.instance_, tag, length2 - length1));
297     }
298   };
299 
300   traverse.rowEndTag = function(label, offset, length1, length2) {
301     var tag = new xrx.token.EndTag(label.clone(), offset, length1);
302 
303     if (length1 !== length2) {
304       self.eventNode(new xrx.node.TextS(self.instance_, tag, length2 - length1));
305     }
306     if (label.sameAs(stop)) traverse.stop();
307   };
308 
309   traverse.forward(self.tag_);
310 };
