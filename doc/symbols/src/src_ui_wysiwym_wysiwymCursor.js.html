 22 /**
 23  * Returns whether the current cursor is a selection or a caret.
 24  *
 25  * @param {!xrx.wysiwym.richxml} wysiwym The WYSIWYM control.
 26  * @return {boolean} True if something is selected.

 95 /**
 96  * Returns the current not-tag token which belongs to the left cursor edge.
 97  *
 98  * @param {!xrx.wysiwym.richxml} wysiwym The WYSIWYM control.
 99  * @return {xrx.token.NotTag} The not-tag token.
100  */
101 xrx.wysiwym.cursor.leftNotTag = function(wysiwym) {
102   var tok;
103   var cursor = xrx.wysiwym.cursor;
104   var pilot = wysiwym.getNode().getInstance().getPilot();
105   var leftTokenOutside = cursor.leftTokenOutside(wysiwym).state.context.token;
106   var leftTokenInside = cursor.leftTokenInside(wysiwym).state.context.token;
107 
108   if (leftTokenOutside.type() === xrx.token.NOT_TAG) {
109     tok = leftTokenOutside;
110   } else if (leftTokenInside.type() === xrx.token.NOT_TAG) {
111     tok = leftTokenInside;
112   } else {
113     var label = leftTokenOutside.label().clone();
114     if (leftTokenOutside.type() === xrx.token.START_TAG) {
115       label.push(0);
116     };
117     tok = new xrx.token.NotTag(label);
118   }
119 
120   // TODO: this.getNodeElement().getToken()
121   return pilot.location(wysiwym.getNode().getToken(), tok);
122 };

126 /**
127  * Returns the offset relative to the current not-tag for the
128  * cursor caret or for the left edge of the cursor selection if
129  * something is selected.
130  * 
131  * @param {!xrx.wysiwym.richxml} wysiwym The WYSIWYM control.
132  * @return {integer} The offset.
133  */
134 xrx.wysiwym.cursor.leftNotTagOffset = function(wysiwym) {
135   var placeholder = xrx.wysiwym.richxml.placeholder;
136   var index = xrx.wysiwym.cursor.leftIndex(wysiwym) - 1;
137   var text = wysiwym.getValue();
138   var offset = 0;
139 
140   for(var i = index; i >= 0; i--) {
141     if (text[i] === placeholder.startTag) break;
142     if (text[i] === placeholder.endTag) break;
143     if (text[i] === placeholder.emptyTag) break;
144     offset++;
145   };
146 
147   return offset;
148 };

152 /**
153  * Returns whether the cursor caret or the left edge of a cursor
154  * selection is placed at the very beginning of a WYSIWYM control.
155  *
156  * @param {!xrx.wysiwym.richxml} wysiwym The WYSIWYM control.
157  * @return {boolean} True if the cursor is placed at the beginning
158  * otherwise false.
159  */
160 xrx.wysiwym.cursor.leftAtStartPosition = function(wysiwym) {
161   var pos = xrx.wysiwym.cursor.leftPosition(wysiwym);
162 
163   return pos.line === 0 && pos.ch === 0;
164 };

168 /**
169  * Returns whether the cursor caret or the left edge of a cursor
170  * selection is placed at the very end of a WYSIWYM control.
171  *
172  * @param {!xrx.wysiwym.richxml} wysiwym The WYSIWYM control.
173  * @return {boolean} True if the cursor is placed at the end
174  * otherwise false.
175  */
176 xrx.wysiwym.cursor.leftAtEndPosition = function(wysiwym) {
177   var pos = xrx.wysiwym.cursor.leftPosition(wysiwym);
178   var cm = wysiwym.codemirror_;
179   var last = cm.lineCount() - 1;
180   var line = cm.getLine(last);
181 
182   return pos.line === last && pos.ch === line.length;
183 };
