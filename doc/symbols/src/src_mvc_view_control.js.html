 16 /**
 17  * @constructor
 18  */
 19 xrx.control = function(element) {
 20   goog.base(this, element);
 21 };
 22 goog.inherits(xrx.control, xrx.component);

 25 
 26 xrx.control.prototype.getRepeat = function() {
 27   var element = goog.dom.getAncestorByClass(this.element_, 'xrx-repeat');
 28   var id = element.getAttribute('id');
 29 
 30   return xrx.view.getComponent(id);
 31 };
 32 
 33 
 34 
 35 xrx.control.prototype.getRepeatIndex = function() {
 36   var repeatItem = goog.dom.getAncestorByClass(this.element_,
 37       'xrx-repeat-item');
 38 
 39   if (goog.dom.classes.has(this.element_, 'xrx-repeat-item')) {
 40 
 41     return this.element_.getAttribute('data-xrx-repeat-index');
 42 
 43   } else if (repeatItem) {
 44 
 45     return repeatItem.getAttribute('data-xrx-repeat-index');
 46 
 47   } else {
 48 
 49     throw Error('Repeat item could not be found.');
 50   }
 51 };
 52 
 53 
 54 
 55 xrx.control.prototype.getNodeBind = function(num) {
 56   return this.getBind().node_[num];
 57 };
 58 
 59 
 60 
 61 xrx.control.prototype.getNodeRef = function() {
 62   var context = this.getRepeat().getNode(this.getRepeatIndex());
 63   // TODO: Node conversion function
 64   var nodeS = new xrx.node.ElementS(context.getInstance(), context.getToken());
 65   var result = xrx.xpath.evaluate(this.getRefExpression(), nodeS, null,
 66       xrx.xpath.XPathResultType.ANY_TYPE);
 67 
 68   return context.isSameAs(nodeS) ? context : result.iterateNext();
 69 };

 73 /**
 74  * Gets the node referenced by the control.
 75  * @return {xrx.node} The node.
 76  */
 77 xrx.control.prototype.getNode = function(num) {
 78   var n = num || 0;
 79 
 80   if (this.getBind()) {
 81 
 82     return this.getNodeBind(n);
 83 
 84   } else if (this.getRefExpression()) {
 85 
 86     return this.getNodeRef(n);
 87 
 88   } else {
 89     throw Error('A control must define a data-xrx-bind or a data-xrx-ref ' +
 90         'attribute.');
 91   }
 92 };
