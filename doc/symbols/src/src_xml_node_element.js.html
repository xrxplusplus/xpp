 18 /**
 19  * Indicates whether two nodes are the same.
 20  *
 21  * @param {!xrx.node} node The node to test against.
 22  * @return {boolean} Whether the nodes are the same.
 23  */
 24 xrx.node.Element.prototype.isSameAs = function(node) {
 25   return this.getType() === node.getType() && 
 26       this.getLabel().sameAs(node.getLabel());
 27 };

 31 /**
 32  * Indicates whether a node appears before another node
 33  * in document order.
 34  * 
 35  * @param {!xrx.node} node The node to test against.
 36  * @return {boolean}
 37  */
 38 xrx.node.Element.prototype.isBefore = function(node) {
 39   var selfLabel = this.getLabel();
 40   var nodeLabel = node.getLabel();
 41 
 42   return selfLabel.isBefore(nodeLabel) ||
 43       ( selfLabel.sameAs(nodeLabel) &&
 44           this.getType() < node.getType() );
 45 };

131 xrx.node.Element.prototype.getNodeAncestor = function(test) {
132   var nodeset = this.find(test, xrx.node[this.impl_.Element].prototype.isDescendantOf,
133       true, new xrx.label());
134 
135   // TODO: not sure if this is correct?
136   if (test.getName() === 'node') 
137       nodeset.unshift(new xrx.node[this.impl_.Document](this.instance_));
138   return nodeset;
139 };
