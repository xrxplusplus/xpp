139 xrx.node.ElementB.prototype.isParentOf = function(node) {
140   if (node.getType() === xrx.node.ELEMENT) {
141     return this.getKey() === node.getRow().getParent() && node.getKey() !== 0;
142   } else {
143     return this.getLabel().isParentOf(node.getLabel());
144   }

166 xrx.node.ElementB.prototype.getName = function() {
167   var inst = this.instance_;
168   var tag = inst.getIndex().getTag(this.key_);
169   var loc = inst.getStream().tagName(tag.xml(inst.xml()));
170   loc.offset += tag.offset();
171 
172   return loc.xml(inst.xml());
173 };

189 /**
190  * 
191  */
192 xrx.node.ElementB.prototype.getStringValue = function() {
193   if (this.getRow().getType() === xrx.token.EMPTY_TAG) return '';
194 
195   var string = '';
196   var xml = this.instance_.xml();
197   var row;
198   var selfLabel = this.getLabel();
199 
200   for(var key = this.getKey(); key <= this.getIndex().getLastKey(); key++) {
201     row = this.getIndex().getRowByKey(key);
202     if (row.getType() === xrx.token.END_TAG && 
203         this.getIndex().getLabel(key).sameAs(selfLabel)) break;
204     string += xml.substr(row.getOffset() + row.getLength1(),
205         row.getLength2() - row.getLength1());
206   };
207 
208   return string;
209 };

213 /**
214  * 
215  */
216 xrx.node.ElementB.prototype.getXml = function() {
217 
218   if (this.getRow().getType() === xrx.token.EMPTY_TAG) {
219 
220     return this.instance_.xml().substr(this.getOffset(), this.getLength());
221   } else {
222     var row = this.getIndex().getRowByTag(new xrx.token.EndTag(
223         this.getLabel()), this.getKey());
224 
225     return this.instance_.xml().substring(this.getOffset(), row.getOffset() +
226         row.getLength1());
227   }
228 };

246   var locs = this.instance_.getStream().attributes(xml);
247   var i = 0;

251     var attr = new xrx.node.AttributeB(parseInt(l), this);

309 /**
310  * @param {!xrx.label}
311  */
312 xrx.node.ElementB.prototype.forward = function(stop, needTextNode) {
313   var self = this;
314   var index = this.instance_.getIndex();
315   index.iterSetKey(this.key_);
316   var row = index.iterGetRow();
317   var type;
318 
319   do {
320     type = row.getType();
321 
322     switch(type) {
323     case xrx.token.START_TAG:
324       self.eventNode(new xrx.node.ElementB(self.instance_, index.iterGetKey()));
325       break;
326     case xrx.token.EMPTY_TAG:
327       self.eventNode(new xrx.node.ElementB(self.instance_, index.iterGetKey()));
328       break;
329     default:
330       break;
331     };
332 
333     if (needTextNode && row.getLength1() !== row.getLength2()) {
334       self.eventNode(new xrx.node.TextB(self.instance_, index.iterGetKey()));
335     }
336 
337     if (type === xrx.token.END_TAG &&
338         self.getIndex().getLabel(index.iterGetKey()).sameAs(stop)) break;
339 
340   } while (row = index.iterNext());
341 };

345 /**
346  * @param {!xrx.label}
347  */
348 xrx.node.ElementB.prototype.backward = function(stop, needTextNode) {
349   var self = this;
350   var index = this.getIndex();
351   index.iterSetKey(this.key_);
352   var row = index.iterGetRow();
353   var type;
354 
355   do {
356     type = row.getType();
357 
358     if (needTextNode && row.getLength1() !== row.getLength2()) {
359       self.eventNode(new xrx.node.TextB(self.instance_, index.iterGetKey()));
360     }
361 
362     switch(type) {
363     case xrx.token.START_TAG:
364       self.eventNode(new xrx.node.ElementB(self.instance_, index.iterGetKey()));
365       break;
366     case xrx.token.EMPTY_TAG:
367       self.eventNode(new xrx.node.ElementB(self.instance_, index.iterGetKey()));
368       break;
369     default:
370       break;
371     };
372 
373     if (type === xrx.token.END_TAG &&
374         self.getIndex().getLabel(index.iterGetKey()).sameAs(stop)) break;
375 
376   } while (row = index.iterPrevious());
377 };
