  9 goog.require('goog.dom');
 10 goog.require('goog.events');

 12 goog.require('xrx.model');

 14 goog.require('xrx.pilot');

 18 /**
 19  * @constructor
 20  */
 21 xrx.instance = function(element) {
 22   goog.base(this, element);
 23 
 24 
 25 
 26   this.xml_;
 27 
 28   this.recalculate();
 29   goog.dom.setTextContent(this.getElement(), '');
 30 };
 31 goog.inherits(xrx.instance, xrx.model);
 32 
 33 
 34 
 35 /**
 36  * 
 37  */
 38 xrx.instance.prototype.getDataInline = function() {
 39   this.xml_ = goog.dom.getRawTextContent(this.getElement());
 40 };
 41 
 42 
 43 
 44 /**
 45  * 
 46  */
 47 xrx.instance.prototype.getDataRemote = function() {
 48 
 49   var request = new goog.net.XhrIo();
 50 
 51   goog.events.listen(request, 'complete', function() {
 52     if(request.getResponseHeader('Content-Type') != 'text/plain') 
 53         throw Error('Expected content type is "text/plain."');
 54     this.xml_ = request.getResponseText();
 55   });
 56 
 57   request.send(this.getSrcUri(), 'GET');
 58 };
 59 
 60 
 61 
 62 
 63 /**
 64  * @override
 65  */
 66 xrx.instance.prototype.recalculate = function() {
 67   this.getSrcUri() ? this.getDataRemote() : this.getDataInline();
 68 };

 72 /**
 73  * @return {!string} The XML instance as string.
 74  */
 75 xrx.instance.prototype.xml = function(xml) {
 76   if (xml) this.xml_ = xml;
 77   return this.xml_;
 78 };
 79 
 80 
 81 
 82 /**
 83  * @return {!xrx.node.Document} The XML instance as node.
 84  */
 85 xrx.instance.prototype.document = function(id) {
 86   var pilot = new xrx.pilot(this.xml());
 87   var node = new xrx.node.Document(pilot, this.getId());
 88   
 89   return node;
 90 };
