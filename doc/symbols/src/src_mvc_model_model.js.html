 62 
 63 xrx.model.getInstanceDefault = function() {
 64   var instance;
 65 
 66   for(var i in xrx.model.components_) {
 67     var component = xrx.model.components_[i];
 68     if (component instanceof xrx.instance) {
 69       instance = component;
 70       break;
 71     }
 72   }
 73 
 74   return instance;
 75 };
