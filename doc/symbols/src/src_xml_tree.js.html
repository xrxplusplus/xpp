109 /**
110  * 
111  */
112 xrx.tree.prototype.forward = function() {
113   var tree = this;
114   var label = new xrx.label();
115   var lastTag;
116 
117   this.stream_.rowStartTag = function(offset, length1, length2) {
118     !lastTag || lastTag === xrx.token.START_TAG ? label.child() : label.nextSibling();
119     tree.rowStartTag(label.clone(), offset, length1, length2);
120     lastTag = xrx.token.START_TAG;
121   };
122 
123   this.stream_.rowEmptyTag = function(offset, length1, length2) {
124     !lastTag || lastTag === xrx.token.START_TAG ? label.child() : label.nextSibling();
125     tree.rowEmptyTag(label.clone(), offset, length1, length2);
126     lastTag = xrx.token.END_TAG;
127   };
128 
129   this.stream_.rowEndTag = function(offset, length1, length2) {
130     if (lastTag !== xrx.token.START_TAG) label.parent();
131     tree.rowEndTag(label.clone(), offset, length1, length2);
132     lastTag = xrx.token.END_TAG;
133   };
134 
135   this.stream_.namespace = function(offset, length1, length2) {
136     tree.namespace(label.clone());
137   };
138   
139   this.stream_.forward();
140 };
