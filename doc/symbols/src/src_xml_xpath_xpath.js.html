 60  * Adds a new prefix and URI to the static XPath context.
 61  * 
 62  * @param {!string} prefix The namespace prefix.
 63  * @param {!string} uri The namespace uri.
 64  */
 65 xrx.xpath.declareNamespace = function(prefix, uri) {
 66   xrx.xpath.XPathNSResolver[prefix] = uri;

 71 /**
 72  * Adds a custom function to the static XPath context.
 73  * 
 74  * @param {!xrx.xpath.Function} The function.
 75  */
 76 xrx.xpath.declareFunction = function(func) {
 77   var prefix = func.name.substr(0, func.name.indexOf(':'));
 78 
 79   if (prefix === '') throw Error('The empty namespace is not allowed for custom functions.');
 80   if (!xrx.xpath.XPathNSResolver[prefix]) throw Error('Namespace <' + prefix + '> is not defined.');
 81 
 82   if (xrx.xpath.XPathNSResolver[func.name]) throw Error('Function <' + func.name + '> is already defined.');
 83   xrx.xpath.XPathNSResolver[func.name] = func;
 84 
 85   xrx.xpath.FunctionCall.createFunc(func.name, func.retrunType,
 86       true, true, true, func.evaluate, func.minArgs, func.maxArgs);
 87 };

 91 /**
 92  * Enum for XPathResult types.
 93  *
 94  * @private
 95  * @enum {number}
 96  */
 97 xrx.xpath.XPathResultType = {
 98   ANY_TYPE: 0,
 99   NUMBER_TYPE: 1,
100   STRING_TYPE: 2,
101   BOOLEAN_TYPE: 3,
102   UNORDERED_NODE_ITERATOR_TYPE: 4,
103   ORDERED_NODE_ITERATOR_TYPE: 5,
104   UNORDERED_NODE_SNAPSHOT_TYPE: 6,
105   ORDERED_NODE_SNAPSHOT_TYPE: 7,
106   ANY_UNORDERED_NODE_TYPE: 8,
107   FIRST_ORDERED_NODE_TYPE: 9
108 };

112 /**
113  * The exported XPathExpression type.
114  *
115  * @constructor
116  * @extends {XPathExpression}
117  * @param {string} expr The expression string.
118  * @param {?(XPathNSResolver|function(string): ?string)} nsResolver
119  *     XPath namespace resolver.
120  * @private
121  */
122 xrx.xpath.XPathExpression = function(expr) {
123   if (!expr.length) {
124     throw Error('Empty XPath expression.');
125   }
126   var lexer = xrx.xpath.Lexer.tokenize(expr);
127   if (lexer.empty()) {
128     throw Error('Invalid XPath expression.');
129   }
130 
131   var gexpr = new xrx.xpath.Parser(lexer).parseExpr();
132   if (!lexer.empty()) {
133     throw Error('Bad token: ' + lexer.next());
134   }
135   this['evaluate'] = function(node, type) {
136     var value = gexpr.evaluate(new xrx.xpath.Context(node));
137     return new xrx.xpath.XPathResult(value, type);
138   };
139 };

143 /**
144  * The exported XPathResult type.
145  *
146  * @constructor
147  * @extends {XPathResult}
148  * @param {(!xrx.xpath.NodeSet|number|string|boolean)} value The result value.
149  * @param {number} type The result type.
150  * @private
151  */
152 xrx.xpath.XPathResult = function(value, type) {
153   if (type == xrx.xpath.XPathResultType.ANY_TYPE) {
154     if (value instanceof xrx.xpath.NodeSet) {
155       type = xrx.xpath.XPathResultType.UNORDERED_NODE_ITERATOR_TYPE;
156     } else if (typeof value == 'string') {
157       type = xrx.xpath.XPathResultType.STRING_TYPE;
158     } else if (typeof value == 'number') {
159       type = xrx.xpath.XPathResultType.NUMBER_TYPE;
160     } else if (typeof value == 'boolean') {
161       type = xrx.xpath.XPathResultType.BOOLEAN_TYPE;
162     } else {
163       throw Error('Unexpected evaluation result.');
164     }
165   }
166   if (type != xrx.xpath.XPathResultType.STRING_TYPE &&
167       type != xrx.xpath.XPathResultType.NUMBER_TYPE &&
168       type != xrx.xpath.XPathResultType.BOOLEAN_TYPE &&
169       !(value instanceof xrx.xpath.NodeSet)) {
170     throw Error('value could not be converted to the specified type');
171   }
172   this['resultType'] = type;
173   var nodes;
174   switch (type) {
175     case xrx.xpath.XPathResultType.STRING_TYPE:
176       this['stringValue'] = (value instanceof xrx.xpath.NodeSet) ?
177           value.string() : '' + value;
178       break;
179     case xrx.xpath.XPathResultType.NUMBER_TYPE:
180       this['numberValue'] = (value instanceof xrx.xpath.NodeSet) ?
181           value.number() : +value;
182       break;
183     case xrx.xpath.XPathResultType.BOOLEAN_TYPE:
184       this['booleanValue'] = (value instanceof xrx.xpath.NodeSet) ?
185           value.getLength() > 0 : !!value;
186       break;
187     case xrx.xpath.XPathResultType.UNORDERED_NODE_ITERATOR_TYPE:
188     case xrx.xpath.XPathResultType.ORDERED_NODE_ITERATOR_TYPE:
189     case xrx.xpath.XPathResultType.UNORDERED_NODE_SNAPSHOT_TYPE:
190     case xrx.xpath.XPathResultType.ORDERED_NODE_SNAPSHOT_TYPE:
191       var iter = value.iterator();
192       nodes = [];
193       for (var node = iter.next(); node; node = iter.next()) {
194         nodes.push(node);
195       }
196       this['snapshotLength'] = value.getLength();
197       this['invalidIteratorState'] = false;
198       break;
199     case xrx.xpath.XPathResultType.ANY_UNORDERED_NODE_TYPE:
200     case xrx.xpath.XPathResultType.FIRST_ORDERED_NODE_TYPE:
201       var firstNode = value.getFirst();
202       this['singleNodeValue'] = firstNode;
203       break;
204     default:
205       throw Error('Unknown XPathResult type.');
206   }
207   var index = 0;
208   this['iterateNext'] = function() {
209     if (type != xrx.xpath.XPathResultType.UNORDERED_NODE_ITERATOR_TYPE &&
210         type != xrx.xpath.XPathResultType.ORDERED_NODE_ITERATOR_TYPE) {
211       throw Error('iterateNext called with wrong result type');
212     }
213     return (index >= nodes.length) ? null : nodes[index++];
214   };
215   this['snapshotItem'] = function(i) {
216     if (type != xrx.xpath.XPathResultType.UNORDERED_NODE_SNAPSHOT_TYPE &&
217         type != xrx.xpath.XPathResultType.ORDERED_NODE_SNAPSHOT_TYPE) {
218       throw Error('snapshotItem called with wrong result type');
219     }
220     return (i >= nodes.length || i < 0) ? null : nodes[i];
221   };
222 };

226 xrx.xpath.evaluate = function(expr, context, nsResolver, type, result) {
227   return new xrx.xpath.XPathExpression(expr, nsResolver).
228       evaluate(context, type);
