 10 /**
 11  * Constructs a new label.
 12  *
 13  * @constructor
 14  * @param {?Array.<!number>} array The array of numbers.
 15  */
 16 xrx.label = function(array) {
 17   this.label_ = array || [];
 18 };

 85 /**
 86  * Returns a copy of the label.
 87  * @return {!xrx.label}
 88  */
 89 xrx.label.prototype.clone = function() {
 90   var length = this.length();
 91   var array = new Array(length);
 92 
 93   for(var i = 0; i < length; i++) {
 94     array[i] = this.label_[i];
 95   }
 96   return new xrx.label(array);
 97 };

101 /**
102  * Returns the joint parent of two labels.
103  * @param {!xrx.label} label
104  * @return {!xrx.label}
105  */
106 xrx.label.prototype.jointParent = function(label) {
107   var arr = [];
108 
109   for(var i = 0; i < label.length(); i++) {
110     val = this.label_[i];
111     val === label.value(i) ? arr.push(val) : null;
112   }
113   return arr.length === 0 ? new xrx.label() : new xrx.label(arr);
114 };

163 /**
164  * Indicates whether two labels are the same.
165  * @param {!xrx.label}
166  * @return {!boolean}
167  */
168 xrx.label.prototype.sameAs = function(label) {
169 
170   if (this.label_.length !== label.length()) {
171     return false;
172   }
173   for(var i = 0; i < label.length(); i++) {
174     if (this.label_[i] !== label.value(i)) {
175       return false;
176     }
177   }
178   return true;
179 };

193 /**
194  * Tests whether the label appears before another label
195  * in document order.
196  * @return {!boolean} 
197  */
198 xrx.label.prototype.isBefore = function(label) {
199 
200   for(var i = 0; i < this.length(); i++) {
201     if (this.label_[i] < label.value(i)) return true;
202   }
203   if (this.length() < label.length()) return true;
204   return false;
205 };

275 xrx.label.prototype.isPrecedingSiblingOf = function(label) {
276 
277   if (this.length() !== label.length()) return false;
278   var len = this.length();
279   for (var i = 0; i < len - 1; i++) {
280     if (this.label_[i] !== label.value(i)) return false;
281   }
282   if (this.label_[len - 1] >= label.value(len - 1)) return false;
283   return true;
284 };

286 xrx.label.prototype.isFollowingSiblingOf = function(label) {
287 
288   if (this.length() !== label.length()) return false;
289   var len = this.length();
290   for (var i = 0; i < len - 1; i++) {
291     if (this.label_[i] !== label.value(i)) return false;
292   }
293   if (this.label_[len - 1] <= label.value(len - 1)) return false;
294   return true;
295 };
