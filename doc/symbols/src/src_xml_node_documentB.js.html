107 xrx.node.DocumentB.prototype.getStringValue = function() {
108   var string = '';
109   var xml = this.instance_.xml();
110   var row;
111 
112   for(var key = 0; key <= this.getIndex().getLastKey(); key++) {
113     row = this.getIndex().getRowByKey(key);
114     string += xml.substr(row.getOffset() + row.getLength1(),
115         row.getLength2() - row.getLength1());
116   };
117 
118   return string;
119 };

135 xrx.node.DocumentB.prototype.getNodeChild = function(test) {
136   var nodeset = new xrx.xpath.NodeSet();
137   var element = new xrx.node.ElementB(this.instance_, 0);
138 
139   if (test.matches(element)) nodeset.add(element);
140 
141   return nodeset;
142 };

146 xrx.node.DocumentB.prototype.getNodeDescendant = function(test) {
147   var nodeset = new xrx.xpath.NodeSet();
148   var index = this.instance_.getIndex();
149   index.iterSetKey(0);
150   var row = index.iterGetRow();
151   var element;
152   var text;
153   var needTextNode = test.needsTextNode();
154 
155   do {
156 
157     if (row.getType() !== xrx.token.END_TAG) {
158       element = new xrx.node.ElementB(this.instance_, index.iterGetKey());
159       if (test.matches(element)) {
160         nodeset.add(element);
161       }
162     }
163 
164     if (needTextNode && row.getLength1() !== row.getLength2()) {
165       text = new xrx.node.TextB(this.instance_, index.iterGetKey());
166       if (test.matches(text)) {
167         nodeset.add(text);
168       }
169     }
170  
171   } while(row = index.iterNext());
172 
173   return nodeset;
174 };
