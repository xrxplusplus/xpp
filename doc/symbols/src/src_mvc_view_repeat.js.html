 15 xrx.repeat = function(element) {
 16   goog.base(this, element);
 17 };
 18 goog.inherits(xrx.repeat, xrx.view);
 19 
 20 
 21 
 22 xrx.component.prototype.createDom = function() {
 23   var child = goog.dom.getFirstElementChild(this.element_);
 24   var n = 0;
 25   var node;
 26 
 27   while(node = this.getNode(n)) {
 28     if (n === 0) {
 29 
 30       goog.dom.setProperties(child, {'data-xrx-repeat-index': n});
 31       goog.dom.classes.add(child, 'xrx-repeat-item');
 32     } else {
 33       var newElement = child.cloneNode(true);
 34 
 35       goog.dom.setProperties(newElement, {'data-xrx-repeat-index': n});
 36       goog.dom.classes.add(newElement, 'xrx-repeat-item');
 37       goog.dom.appendChild(this.element_, newElement);
 38     }
 39     n++;
 40   }
 41 };
