 23 /**
 24  * @constructor
 25  */
 26 xrx.instance = function(element, callback) {
 27   goog.base(this, element);
 28 
 29 
 30 
 31   this.xml_;
 32 
 33 
 34 
 35   this.stream_;
 36 
 37 
 38 
 39   this.pilot_;
 40 
 41 
 42 
 43   this.recalculate(callback);
 44 };

 49 /**
 50  * 
 51  */
 52 xrx.instance.prototype.update = function(offset, length, xml) {
 53 
 54   var tmp = this.xml_.substr(0, offset) + xml + this.xml_.substr(
 55       offset + length);
 56 
 57   this.xml_ = undefined;
 58   this.xml_ = tmp;
 59   this.stream_ = undefined;
 60   this.stream_ = new xrx.stream(this.xml_);
 61   this.pilot_ = undefined;
 62   this.pilot_ = new xrx.pilot(this.xml_);
 63 };

 67 /**
 68  * 
 69  */
 70 xrx.instance.prototype.getDataInline = function() {
 71   var parse = new xrx.parse();
 72 
 73   this.xml_ = parse.normalize(goog.dom.getRawTextContent(this.getElement()));
 74   this.stream_ = new xrx.stream(this.xml_);
 75   this.pilot_ = new xrx.pilot(this.xml_);
 76 };
 77 
 78 
 79 
 80 /**
 81  * 
 82  */
 83 xrx.instance.prototype.getDataRemote = function(callback) {
 84   var self = this;
 85   var request = new goog.net.XhrIo();
 86   var parse = new xrx.parse();
 87 
 88   goog.events.listen(request, 'complete', function() {
 89     if(request.getResponseHeader('Content-Type') != 'text/plain') 
 90         throw Error('Expected content type is "text/plain."');
 91     self.xml_ = parse.normalize(request.getResponseText());
 92     self.stream_ = new xrx.stream(self.xml_);
 93     self.pilot_ = new xrx.pilot(this.xml_);
 94 
 95     if (callback) callback(self);
 96   });
 97 
 98   request.send(this.getSrcUri(), 'GET');
 99 };

104 /**
105  * @override
106  */
107 xrx.instance.prototype.recalculate = function(callback) {
108   this.getSrcUri() ? this.getDataRemote(callback) : this.getDataInline();
109 };
