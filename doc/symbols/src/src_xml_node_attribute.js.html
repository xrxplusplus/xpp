 30 xrx.node.Attribute.prototype.isSameAs = function(node) {
 31   return this.getType() === node.getType() && 
 32       this.getLabel().sameAs(node.getLabel());
 33 };

 37 xrx.node.Attribute.prototype.isBefore = function(node) {
 38   var selfLabel = this.getLabel();
 39   var nodeLabel = node.getLabel();
 40 
 41   return selfLabel.isBefore(nodeLabel) ||
 42       ( selfLabel.sameAs(nodeLabel) &&
 43           this.getType() < node.getType() );
 44 };

 47 
 48 xrx.node.Attribute.prototype.isAfter = function(node) {
 49   var selfLabel = this.getLabel();
 50   var nodeLabel = node.getLabel();
 51 
 52   return selfLabel.isAfter(nodeLabel) ||
 53       ( selfLabel.sameAs(nodeLabel) &&
 54           this.getType() > node.getType() );
 55 };

113 xrx.node.Attribute.prototype.getNodeAncestor = function(test) {
114   var nodeset = new xrx.xpath.NodeSet();
115   if (test.matches(this.parent_)) nodeset.add(this.parent_);
116   nodeset.add(this.parent_.getNodeAncestor(test));
117 
118   // TODO: not sure if this is correct?
119   if (test.getName() === 'node') 
120       nodeset.unshift(new xrx.node.DocumentB(this.instance_));
121 
122   return nodeset;
123 };
