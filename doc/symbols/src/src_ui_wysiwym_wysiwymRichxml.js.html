 50 /**
 51  *
 52  */
 53 xrx.wysiwym.richxml.placeholder.matches = function(ch) {
 54 
 55   for(var p in xrx.wysiwym.richxml.placeholder) {
 56     if (xrx.wysiwym.richxml.placeholder[p] === ch) {
 57       return true;
 58     }
 59   }
 60 
 61   return false;
 62 };

 92 /**
 93  * @private
 94  */
 95 xrx.wysiwym.richxml.prototype.transformXml_ = function(xml) {
 96   var visualXml = '';
 97   var self = this;
 98   var stream = new xrx.stream(xml);
 99 
100   stream.rowStartTag = function(offset, length1, length2) {
101     visualXml += xrx.wysiwym.richxml.placeholder.startTag;
102     if(length1 !== length2) visualXml += xml.substr(offset + length1, length2 - length1);
103   };
104 
105   stream.rowEndTag = function(offset, length1, length2) {
106     visualXml += xrx.wysiwym.richxml.placeholder.endTag;
107     if(length1 !== length2) visualXml += xml.substr(offset + length1, length2 - length1);
108   };
109 
110   stream.rowEmptyTag = function(offset, length1, length2) {
111     visualXml += xrx.wysiwym.richxml.placeholder.emptyTag;
112     if(length1 !== length2) visualXml += xml.substr(offset + length1, length2 - length1);
113   };
114 
115   stream.forward();
116 
117   return visualXml;
118 };

132 /**
133  *
134  */
135 xrx.wysiwym.richxml.prototype.refresh = function() {
136 
137   if (!this.getNode()) {
138     this.setReadonly(true);
139     return;
140   }
141 
142   if (this.getNode().getType() !== xrx.node.ELEMENT) {
143     throw Error('WYSIWYM controls must be bound to element nodes.');
144   }
145 
146   var visualXml = this.transformXml_(this.getNode().getXml());
147   this.setValue(visualXml, true);
148 };

171 /**
172  *
173  */
174 xrx.wysiwym.richxml.prototype.getCorrespondingPosition = function(token) {
175   var cm = this.codemirror_;
176   var placeholder = xrx.wysiwym.richxml.placeholder;
177   var text = cm.getValue();
178   var index = cm.indexFromPos(cm.getCursor()) - 1;
179   var ch;
180   var stack = 0;
181 
182   if(token.string === placeholder.startTag) {
183 
184     for(var i = index; i < text.length; i++) {
185       ch = text[i];
186       if(ch === placeholder.endTag) stack -= 1;
187       if(ch === placeholder.startTag) stack += 1;
188       if(stack === 0) {
189         index = i + 1;
190         break;
191       } 
192     }
193   } else if(token.string === placeholder.endTag) {
194     console.log('test');
195     for(var i = index; i >= 0; i--) {
196       ch = text[i];
197       if(ch === placeholder.startTag) stack += 1;
198       if(ch === placeholder.endTag) stack -= 1;
199       if(stack === 0) {
200         index = i + 1;
201         break;
202       } 
203     }
204   } else {};
205 
206   return this.codemirror_.posFromIndex(index);
207 };

211 /**
212  *
213  */
214 xrx.wysiwym.richxml.prototype.markActiveTags = function(unmarkflag) {
215   var cm = this.codemirror_;
216   var cursor = xrx.wysiwym.cursor;
217   var placeholder = xrx.wysiwym.richxml.placeholder;
218   var className = xrx.wysiwym.richxml.className;
219   var self = this;
220   
221   var markedElements = function() { 
222     return self.markedElements_ || (self.markedElements_ = []); 
223   };
224   
225   var mark = function(from, to, style) {
226     var mark = cm.markText(from, to, style);
227 
228     markedElements().push(mark);
229   };
230   
231   var unmark = function() {
232 
233     for(var i = 0; i < markedElements().length; ++i) {
234       markedElements()[i].clear();
235     }
236     self.markedElements_ = [];
237   };
238 
239   unmark();
240   if (!this.cursorIsNearTag()) return;
241   if (unmarkflag) return;
242 
243   var leftTokenOutside = cursor.leftTokenOutside(self);
244   var leftPosition = cursor.leftPosition(self);
245   var firstMark = leftTokenOutside.string === placeholder.startTag ? 
246       className.startTagActive : className.endTagActive;
247   var secondMark = leftTokenOutside.string === placeholder.startTag ? 
248       className.endTagActive : className.startTagActive;
249 
250   // mark matched element
251   mark(
252       { line: leftPosition.line, ch: leftPosition.ch - 1 }, 
253       { line: leftPosition.line, ch: leftPosition.ch }, 
254       { className: firstMark }
255   );
256   
257   if (leftTokenOutside.string === placeholder.emptyTag) return;
258   
259   // mark corresponding element
260   var corresponding = this.getCorrespondingPosition(leftTokenOutside);
261   mark(
262       { line: corresponding.line, ch: corresponding.ch - 1 },
263       { line: corresponding.line, ch: corresponding.ch },
264       { className: secondMark }
265   );
266   
267   // mark text
268   var from = { line: leftPosition.line, ch: leftPosition.ch };
269   var to = { line: corresponding.line, ch: corresponding.ch - 1 };
270   leftTokenOutside.string === placeholder.startTag ? 
271       mark(from, to, { className: className.notTagActive }) : 
272       mark(to, from, { className: className.notTagActive });
273 };

286 /**
287  * Does the necessary operations after the cursor selection has moved.
288  */
289 xrx.wysiwym.richxml.prototype.doCursorActivitySelection = function() {
290   var cursor = xrx.wysiwym.cursor;
291 
292   // start-tag at the very beginning or end-tag at the very end selected?
293   if (cursor.leftAtStartPosition(this) || cursor.rightAtEndPosition(this)) {
294     this.setStateInvalid(true);
295     return;
296   }
297 
298   // end-tag inside the left cursor edge or start-tag inside the right cursor edge?
299   var leftTokenInside = cursor.leftTokenInside(this).state.context.token;
300   var rightTokenInside = cursor.rightTokenInside(this).state.context.token;
301 
302   if (leftTokenInside.type() === xrx.token.END_TAG ||
303       rightTokenInside.type() === xrx.token.START_TAG) {
304     this.setStateInvalid(true);
305     return;
306   }
307 
308   // left token inside and right token inside have the same parent token?
309   var leftLabel = leftTokenInside.label().clone();
310   var rightLabel = rightTokenInside.label().clone();
311   leftLabel.parent();
312   rightLabel.parent();
313 
314   if (!leftLabel.sameAs(rightLabel)) {
315     this.setStateInvalid(true);
316   }
317 };

361  * @return {boolean} True if content is removed, otherwise false.

363 xrx.wysiwym.richxml.prototype.isSomethingRemoved = function(change) {
364   return change.text[0] === '';

369 /**
370  * Performs the necessary model updates before something is inserted.
371  */
372 xrx.wysiwym.richxml.prototype.doSomethingInserted = function(change) {
373   var pilot = this.getNode().getInstance().getPilot();
374   var notTag = xrx.wysiwym.cursor.leftNotTag(this);
375 
376   // TODO: multi-line support
377   if (!xrx.wysiwym.cursor.isSomethingSelected(this)) {  
378     xrx.controller.insertNotTag(this, notTag, xrx.wysiwym.cursor.leftNotTagOffset(this),
379         change.text.join('\n'));
380   } else {
381     change.cancel();
382   }
383 };

387 /**
388  * Performs the necessary model updates and internal control updates
389  * before something is removed.
390  */
391 xrx.wysiwym.richxml.prototype.doSomethingRemoved = function(change) {
392   var cm = this.codemirror_;
393   var placeholder = xrx.wysiwym.richxml.placeholder;
394   var pilot = this.getNode().getInstance().getPilot();
395   var removed = this.getValue().substring(change.from.ch, change.to.ch);
396 
397   // TODO: multi-line support
398 
399   // one character removed?
400   if (!xrx.wysiwym.cursor.isSomethingSelected(this)) {
401 
402     // attempt to remove the very first character?
403     if (change.from.line === 0 &&
404         change.from.ch === 0) {
405       change.cancel();
406       return;
407     }
408 
409     // attempt to remove the very last character?
410     var lastLine = cm.lastLine();
411     var lastChar = cm.getLine(lastLine).length - 1;
412     if (change.from.line === lastLine &&
413         change.from.ch === lastChar) {
414       change.cancel();
415       return;
416     }
417 
418     // not-tag is removed?
419     if (!placeholder.matches(removed)) {
420       var notTag = xrx.wysiwym.cursor.leftNotTag(this);
421       notTag = pilot.location(this.getNode().getToken(), notTag);
422       var notTagOffset =
423           xrx.wysiwym.cursor.leftNotTagOffset(this) - removed.length;
424 
425       xrx.controller.reduceNotTag(this, notTag, notTagOffset,
426           change.to.ch - change.from.ch);
427     }
428 
429     // tag is removed?
430     else {
431       var token = cm.getTokenAt({line: change.to.line, ch: change.to.ch});
432       var tag = token.state.context.token;
433       tag = pilot.location(this.getNode().getToken(), tag);
434 
435       // empty tag?
436       if (tag.type() === xrx.token.EMPTY_TAG) {
437         xrx.controller.removeEmptyTag(this, tag);
438         return;
439       }
440       // start tag?
441       else if (tag.type() === xrx.token.START_TAG) {
442         var to = this.getCorrespondingPosition(token);
443         var from = {line: to.line, ch: to.ch - 1};
444         var endTag = pilot.location(tag,
445             new xrx.token.EndTag(tag.label().clone()));
446 
447         xrx.controller.removeStartEndTag(this, tag, endTag);
448 
449         this.internalUpdate = true;
450         cm.replaceRange('', from, to);
451         cm.replaceRange('', change.from, change.to);
452         this.internalUpdate = false;
453 
454         change.cancel();
455       }
456       // end tag?
457       else {
458         var to = this.getCorrespondingPosition(token);
459         var from = {line: to.line, ch: to.ch - removed.length};
460         var startTag = pilot.location(this.getNode().getToken(), // TODO: backward piloting
461             new xrx.token.StartTag(tag.label().clone()));
462 
463         xrx.controller.removeStartEndTag(this, startTag, tag);
464 
465         this.internalUpdate = true;
466         cm.replaceRange('', change.from, change.to);
467         cm.replaceRange('', from, to);
468         cm.setCursor({line: change.from.line, ch: change.from.ch - removed.length});
469         this.internalUpdate = false;
470 
471         change.cancel();
472       }
473     }
474   }
475 
476   // selection removed?
477   else { 
478     change.cancel();
479   }
480 };

484 /**
485  * Function is called before the content of the WYSIWYM control changes.
486  *
487  * @param {!CodeMirror} cm The CodeMirror instance.
488  * @param {!Object} change An object containing information about the changes.
489  */
490 xrx.wysiwym.richxml.prototype.eventBeforeChange = function(cm, change) {
491 
492   if (this.internalUpdate === false) {
493 
494     if (xrx.wysiwym.cursor.leftAtStartPosition(this) ||
495         xrx.wysiwym.cursor.leftAtEndPosition(this)) {
496       change.cancel();
497     } else {
498       this.isSomethingRemoved(change) ? this.doSomethingRemoved(change) :
499           this.doSomethingInserted(change);
500     }
501   }
502 };
